Servo Motor Control

Control servo motors for precise angular positioning.

Servo Motor Control:

#include <Servo.h>

Servo myServo;
int pos = 0;

void setup() {
  myServo.attach(9); // Servo on pin 9
}

void loop() {
  // Sweep from 0 to 180 degrees
  for (pos = 0; pos <= 180; pos += 1) {
    myServo.write(pos);
    delay(15);
  }
  // Sweep back
  for (pos = 180; pos >= 0; pos -= 1) {
    myServo.write(pos);
    delay(15);
  }
}

Range: 0-180 degrees. Uses PWM pin (9, 10, or 11)

Ultrasonic Sensor (HC-SR04)

Measure distance using ultrasonic sound waves.

Distance Measurement:

const int trigPin = 9;
const int echoPin = 10;
long duration;
int distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2; // Speed of sound
  
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");
  delay(100);
}

Range: 2cm to 400cm. Uses Trig and Echo pins

I2C Communication

Communicate with multiple devices using I2C protocol.

I2C Communication Example:

#include <Wire.h>

void setup() {
  Wire.begin(); // Join I2C bus
  Serial.begin(9600);
}

void loop() {
  Wire.beginTransmission(0x27); // Device address
  Wire.write(0x00); // Register address
  Wire.endTransmission();
  
  Wire.requestFrom(0x27, 1); // Request 1 byte
  if (Wire.available()) {
    byte data = Wire.read();
    Serial.println(data, HEX);
  }
  delay(500);
}

Pins: SDA (A4), SCL (A5). Supports multiple devices on same bus

SPI Communication:

#include <SPI.h>

  mfrc522.PICC_HaltA();
  delay(1000);
}

7-Segment Display

Display numbers and characters on 7-segment displays.

7-Segment Display Counter:

// Pin definitions for common cathode 7-segment
int segments[] = {2, 3, 4, 5, 6, 7, 8};
int digitPins[] = {9, 10};

// Patterns for digits 0-9
byte digitPatterns[10] = {
  0b00111111, // 0
  0b00000110, // 1
  0b01011011, // 2
  0b01001111, // 3
  0b01100110, // 4
  0b01101101, // 5
  0b01111101, // 6
  0b00000111, // 7
  0b01111111, // 8
  0b01101111  // 9
};

void setup() {
  for (int i = 0; i < 7; i++) {
    pinMode(segments[i], OUTPUT);
  }
  for (int i = 0; i < 2; i++) {
    pinMode(digitPins[i], OUTPUT);
  }
}

void loop() {
  for (int num = 0; num < 100; num++) {
    displayNumber(num);
    delay(500);
  }
}

void displayNumber(int number) {
  int tens = number / 10;
  int ones = number % 10;
  
  // Display tens digit
  digitalWrite(digitPins[0], LOW);
  showDigit(tens);
  delay(5);
  digitalWrite(digitPins[0], HIGH);
  
  // Display ones digit
  digitalWrite(digitPins[1], LOW);
  showDigit(ones);
  delay(5);
  digitalWrite(digitPins[1], HIGH);
}

void showDigit(int digit) {
  byte pattern = digitPatterns[digit];
  for (int i = 0; i < 7; i++) {
    digitalWrite(segments[i], bitRead(pattern, i));
  }
}

Tip: Use multiplexing for multiple digits. Consider using MAX7219 driver IC

Button Input & Debouncing

Learn to read button inputs and handle debouncing.

Button Input with Debouncing:

const int buttonPin = 2;
const int ledPin = 13;
int buttonState = 0;
int lastButtonState = 0;
unsigned long lastDebounceTime = 0;
const unsigned long debounceDelay = 50;

void setup() {
  pinMode(buttonPin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  int reading = digitalRead(buttonPin);
  
  if (reading != lastButtonState) {
    lastDebounceTime = millis();
  }
  
  if ((millis() - lastDebounceTime) > debounceDelay) {
    if (reading != buttonState) {
      buttonState = reading;
      if (buttonState == HIGH) {
        digitalWrite(ledPin, HIGH);
      } else {
        digitalWrite(ledPin, LOW);
      }
    }
  }
  
  lastButtonState = reading;
}

Key Concept: Debouncing prevents false readings from button bouncing

Time & Millis() Function

Master timing in Arduino without using delay().

Non-Blocking Timing Code:

unsigned long previousMillis = 0;

void setup() {
  Serial.begin(9600);

void loop() {

}

  Serial.println("¬∞C");
  
  delay(2000);
}

DHT11: 20-80% RH, 0-50¬∞C. DHT22: 0-100% RH, -40-80¬∞C

}

7-Segment Display

Display numbers and characters on 7-segment displays.

7-Segment Display Counter:

// Pin definitions for common cathode 7-segment
int segments[] = {2, 3, 4, 5, 6, 7, 8};
int digitPins[] = {9, 10};

// Patterns for digits 0-9
byte digitPatterns[10] = {
  0b00111111, // 0
  0b00000110, // 1
  0b01011011, // 2
  0b01001111, // 3
  0b01100110, // 4
  0b01101101, // 5
  0b01111101, // 6
  0b00000111, // 7
  0b01111111, // 8
  0b01101111  // 9
};

void setup() {
  for (int i = 0; i < 7; i++) {
    pinMode(segments[i], OUTPUT);
  }
  for (int i = 0; i < 2; i++) {
    pinMode(digitPins[i], OUTPUT);
  }
}

void loop() {
  for (int num = 0; num < 100; num++) {
    displayNumber(num);
    delay(500);
  }
}

void displayNumber(int number) {
  int tens = number / 10;
  int ones = number % 10;
  
  // Display tens digit
  digitalWrite(digitPins[0], LOW);
  showDigit(tens);
  delay(5);
  digitalWrite(digitPins[0], HIGH);
  
  // Display ones digit
  digitalWrite(digitPins[1], LOW);
  showDigit(ones);
  delay(5);
  digitalWrite(digitPins[1], HIGH);
}

void showDigit(int digit) {
  byte pattern = digitPatterns[digit];
  for (int i = 0; i < 7; i++) {
    digitalWrite(segments[i], bitRead(pattern, i));
  }
}

Tip: Use multiplexing for multiple digits. Consider using MAX7219 driver IC

‚ö° Pinout Tips & Best Practices

Analog Pins as Digital

You can use analog pins as digital pins: pinMode(A0, OUTPUT) and digitalWrite(A0, HIGH)

PWM Pins

Use analogWrite() only on PWM pins for analog output (0-255 maps to 0-5V)

Serial Pins

Avoid using pins 0 & 1 for other purposes when using Serial communication

Current Limits

Each pin can source/sink ~40mA. Total board ~200mA. Use external power for motors

Voltage Levels

Digital pins: 5V (or 3.3V on some boards). Analog input: 0-5V for 0-1023

Protection

Always use current-limiting resistors with LEDs and external circuits
